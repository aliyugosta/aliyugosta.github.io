SQL · Human Resources & Sales Data

SQLQuery1.sql — AdventureWorks2014
-- Sales order analysis: line totals by order SELECT SalesOrderID, SUM(LineTotal) AS total_order_value FROM [AdventureWorks2014].[Sales].[SalesOrderDetail] GROUP BY SalesOrderID ORDER BY total_order_value DESC; -- Employee demographics: female employees born before 1980 SELECT emp.BusinessEntityID, emp.JobTitle, emp.BirthDate, emp.Gender, emp.HireDate, per.AddressLine1, per.City FROM HumanResources.Employee emp LEFT JOIN Person.BusinessEntityAddress ad ON emp.BusinessEntityID = ad.BusinessEntityID LEFT JOIN Person.Address per ON per.AddressID = ad.AddressID WHERE emp.Gender = 'F' AND emp.BirthDate <= '1980-01-01';
